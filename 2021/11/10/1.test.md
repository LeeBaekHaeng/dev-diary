select
    a.CMMNTY_ID ,
    a.EMPLYR_ID ,
    b.USER_NM as EMPLYR_NM ,
    b.USER_ID ,
    b.USER_EMAIL ,
    a.MNGR_AT ,
    a.SBSCRB_DE ,
    a.SECSN_DE ,
    a.MBER_STTUS ,
    c.CODE_NM as MBER_STTUS_NM ,
    a.USE_AT ,
    a.FRST_REGIST_PNTTM ,
    a.FRST_REGISTER_ID
from
    COMTNCMMNTYUSER a
left outer join COMVNUSERMASTER b on
    a.EMPLYR_ID = b.ESNTL_ID
left outer join (
    select
        CODE_ID,
        CODE,
        CODE_NM
    from
        COMTCCMMNDETAILCODE
    where
        CODE_ID = 'COM013'
        and USE_AT = 'Y') c on
    a.MBER_STTUS = c.CODE
where
    1 = 1
    and a.CMMNTY_ID = 'CMMNTY_0000000000211'
    and b.USER_NM like CONCAT ('%',
    '',
    '%')
order by
    b.USER_NM
limit 10 OFFSET 0 ;
