select
    a.NCRD_ID,
    b.NM,
    b.CMPNY_NM,
    b.DEPT_NM,
    b.FRST_REGIST_PNTTM,
    a.EMPLYR_ID,
    a.USE_AT,
    c.USER_NM,
    DATE_FORMAT(a.CREAT_DT, '%Y-%m-%d') as CREAT_DT,
    b.FRST_REGISTER_ID
from
    COMTNNCRDUSER a
left outer join COMTNNCRD b on
    a.NCRD_ID = b.NCRD_ID
left outer join COMVNUSERMASTER c on
    a.EMPLYR_ID = c.ESNTL_ID
where
    a.USE_AT = 'Y'
    and a.EMPLYR_ID = 'USRCNFRM_00000000000'
    and NM like CONCAT ('%',
    'test 이름 2021-12-24 06:50:57',
    '%')
order by
    NM
limit 10 OFFSET 0
;
