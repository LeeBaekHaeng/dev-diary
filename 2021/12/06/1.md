select
    COUNT(a.ESNTL_ID)
from
    COMVNUSERMASTER a,
    COMTNCLUBUSER b
where
    a.ESNTL_ID = b.EMPLYR_ID
    and b.OPRTR_AT = 'Y'
    and b.CLB_ID = 'CLB_0000000000000031'
    and a.USER_NM like CONCAT ('%',
    'T일반회원2021_12_06_20_10_33',
    '%') ;

select * from COMTNCLUBUSER
;
